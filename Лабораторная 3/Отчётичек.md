char bit(int x,int y,int k,int b, BmpImg img) { 
  char a;
  if (k == 0) {
    a=img.red_at(x, y);
  }
  else if (k == 1) {
    a=img.green_at(x, y);
  }
  else if (k == 2) {
    a=img.blue_at(x, y);
  }
  a = a & (~((1 << 1) | (1 << 2) | (1 << 3) | (1 << 4) | (1 << 5) | (1 << 6) | (1 << 7))); // обнуление первых 7ми битов
  a = (a << b); //sdvig
  return a;
}
